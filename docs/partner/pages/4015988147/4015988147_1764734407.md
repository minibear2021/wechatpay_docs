# 1\. 目标

通过本教程的学习，你应该可以：

- 可以[申请所有/单个特约商户交易账单](https://pay.weixin.qq.com/doc/v3/partner/4013080595.md)

- 可以[申请服务商资金账单](https://pay.weixin.qq.com/doc/v3/partner/4013080595.md)

- 可以通过申请结果URL[下载账单](https://pay.weixin.qq.com/doc/v3/partner/4013080597.md)

- 可以正确处理错误码

- 可以校验下载账单内容的摘要值是否正确


# 2\. 业务处理流程

## 2.1. 申请交易账单并下载账单

微信支付在每日10点后生成昨日交易账单文件，商户可通过接口获取账单下载链接。账单包含交易金额、时间及营销信息，利于订单核对、退款审查及银行到账确认。详细介绍参考：[下载账单-产品介绍](https://pay.weixin.qq.com/doc/v3/partner/4013080592.md)。

```
package com.java.downloadbilldemo.partner;

import com.java.demo.DownloadBill; // 下载账单 https://pay.weixin.qq.com/doc/v3/partner/4013080597
import com.java.demo.GetTradeBill; // 申请所有/单个特约商户交易账单 https://pay.weixin.qq.com/doc/v3/partner/4013080595
import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/partner/4014985777

import static com.java.demo.DownloadBill.*; // 下载账单 https://pay.weixin.qq.com/doc/v3/partner/4013080597
import static com.java.demo.GetTradeBill.*; // 申请所有/单个特约商户交易账单 https://pay.weixin.qq.com/doc/v3/partner/4013080595

public class DownloadTradeBillDemo {
    public static void main(String[] args) {
        GetTradeBill getTradeBill = new GetTradeBill(
                "19xxxxxxxx", // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4013080340
                "1DDE55AD98Exxxxxxxxxx", // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4012081992
                "/path/to/apiclient_key.pem", // 商户API证书私钥文件路径，本地文件路径
                "PUB_KEY_ID_xxxxxxxxxxxxx", // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
                "/path/to/wxp_pub.pem" // 微信支付公钥文件路径，本地文件路径
        );

        DownloadBill downloadBill = new DownloadBill(
                "19xxxxxxxx", // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4013080340
                "1DDE55AD98Exxxxxxxxxx", // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4012081992
                "/path/to/apiclient_key.pem", // 商户API证书私钥文件路径，本地文件路径
                "PUB_KEY_ID_xxxxxxxxxxxxx", // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
                "/path/to/wxp_pub.pem" // 微信支付公钥文件路径，本地文件路径
        );
        try {
            GetTradeBillRequest getTradeBillRequest = new GetTradeBillRequest();
            getTradeBillRequest.billDate = "2025-08-11"; // 期望申请下载的账单日期
            getTradeBillRequest.billType = BillType.ALL; // 期望申请下载的账单类型
            getTradeBillRequest.tarType = GetTradeBill.TarType.GZIP; // 期望申请下载的账单类型

            QueryBillEntity response = getTradeBill.run(getTradeBillRequest);

            System.out.println("Download URL: " + response.downloadUrl);

            DownloadBillRequest downloadBillRequest = new DownloadBillRequest();
            downloadBillRequest.downloadUrl = response.downloadUrl;
            downloadBillRequest.localFilePath = "test.csv"; // 期望下载的账单文件路径
            downloadBillRequest.tarType = DownloadBill.TarType.valueOf(getTradeBillRequest.tarType.name());
            downloadBillRequest.expectedHashType = DownloadBill.HashType.valueOf(response.hashType.name());
            downloadBillRequest.expectedHashValue = response.hashValue;

            downloadBill.run(downloadBillRequest);

            System.out.println("File downloaded successfully! Local file path: " + downloadBillRequest.localFilePath);
            // TODO: 请求成功，继续业务逻辑
        } catch (WXPayUtility.ApiException e) {
            // TODO: 请求失败，根据状态码执行不同的逻辑
            String errorCode = e.getErrorCode();
            if (errorCode != null && errorCode.equals("NO_STATEMENT_EXIST")) {
                // 错误：账单不存在
                // 解决方式：请检查当前商户号在指定日期内是否有成功的交易或退款
                // 描述：说明指定日期无账单文件生成。若无交易，则无需重试
            } else if (errorCode != null && errorCode.equals("STATEMENT_CREATING")) {
                // 错误：账单正在生成中
                // 解决方式：请检查当前商户号在指定日期内是否有成功的交易或退款
                // 描述：说明指定日期账单正在生成中。若无交易，则无需重试；若有交易，则在T+1日上午10点后再重新下载
            } else if (errorCode != null && errorCode.equals("SYSTEM_ERROR")) {
                // 错误：系统错误
                // 解决方式：稍后重试
                // 描述：微信支付系统失败，系统失败直接立即重试大概率还会是系统失败，建议等1分钟后再重试
            } else if (errorCode != null && errorCode.equals("FREQUENCY_LIMITED")) {
                // 错误：限频报错
                // 解决方式：稍后重试
                // 描述： 接口频率限制，直接立即重试大概率还会是系统失败，建议等1分钟后再重试
            } else if (errorCode != null && errorCode.equals("SIGN_ERROR")) {
                // 错误：签名错误
                // 解决方式：检查平台商户证书序列号，证书私钥文件，公钥ID，公钥文件，同时确认下签名过程是不是按照微信支付的签名方式
                // 描述：签名报错，需要确认签名材料和签名流程是否正确
            } else if (errorCode != null && errorCode.equals("PARAM_ERROR")) {
                // 错误：参数错误
                // 解决方式：按照报错返回的message，重新输入请求参数
                // 描述：参数的类型，长度，或者必填选项没有填写等
            } else if (errorCode != null && errorCode.equals("INVALID_REQUEST")) {
                // 错误：请求非法，请求参数正确，但是不符合下载账单业务规则
                // 解决方式：根据具体message查看具体哪里不符合业务规则，如果可以修改参数达到符合业务规则的修改请求符合业务规则之后再重试
                // 描述：不符合业务规则的场景如：
                // - 账单时间超过三个月，不允许申请下载账单
            } else {
                // 其他类型错误：稍等一会后重试
            }
        }
    }
}

```

## 2.2. 申请服务商资金账单并下载账单

微信支付按天提供服务商各账户的资金流水账单文件，服务商可以通过该接口获取账单文件的下载地址。账单文件详细记录了服务商账户资金操作的相关信息，包括业务单号、收支金额及记账时间等，以便服务商进行核对与确认。详细介绍参考： [产品介绍](https://pay.weixin.qq.com/doc/v3/partner/4013080592.md)。

```
package com.java.downloadbilldemo.partner;

import com.java.demo.DownloadBill; // 下载账单 https://pay.weixin.qq.com/doc/v3/partner/4013080597
import com.java.demo.DownloadBill.DownloadBillRequest;
import com.java.demo.GetFundFlowBill; // 申请服务商资金账单 https://pay.weixin.qq.com/doc/v3/partner/4013080596
import com.java.demo.GetFundFlowBill.FundFlowBillAccountType;
import com.java.demo.GetFundFlowBill.GetFundFlowBillRequest;
import com.java.demo.GetFundFlowBill.QueryBillEntity;
import com.java.utils.WXPayUtility; // 引用微信支付工具库，参考：https://pay.weixin.qq.com/doc/v3/partner/4014985777

public class DownloadFundFlowBillDemo {

    public static void main(String[] args) {
        String mchCode = "19xxxxxxxx"; // 商户号，是由微信支付系统生成并分配给每个商户的唯一标识符，商户号获取方式参考 https://pay.weixin.qq.com/doc/v3/partner/4012081992
        String mchSerialNo = "1DDE55AD98Exxxxxxxxxx"; // 商户API证书序列号，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4012081992
        String mchPrivateKeyPath = "/path/to/apiclient_key.pem"; // 商户API证书私钥文件路径，本地文件路径
        String mchPublicKeyId = "PUB_KEY_ID_xxxxxxxxxxxxx"; // 微信支付公钥ID，如何获取请参考 https://pay.weixin.qq.com/doc/v3/partner/4013038589
        String mchPublicKeyPath = "/path/to/wxp_pub.pem"; // 微信支付公钥文件路径，本地文件路径

        GetFundFlowBill getFundFlowBill = new GetFundFlowBill(
                mchCode,
                mchSerialNo,
                mchPrivateKeyPath,
                mchPublicKeyId,
                mchPublicKeyPath
        );

        DownloadBill downloadBill = new DownloadBill(
                mchCode,
                mchSerialNo,
                mchPrivateKeyPath,
                mchPublicKeyId,
                mchPublicKeyPath
        );

        try {
            GetFundFlowBillRequest getFundFlowBillRequest = new GetFundFlowBillRequest();
            getFundFlowBillRequest.billDate = "2025-11-25"; // 期望申请下载的账单日期
            getFundFlowBillRequest.accountType = FundFlowBillAccountType.BASIC; // 期望申请下载的账单类型
            getFundFlowBillRequest.tarType = GetFundFlowBill.TarType.GZIP; // 期望申请下载的账单压缩类型

            QueryBillEntity response = getFundFlowBill.run(getFundFlowBillRequest);

            System.out.println("Download URL: " + response.downloadUrl);

            DownloadBillRequest downloadBillRequest = new DownloadBillRequest();
            downloadBillRequest.downloadUrl = response.downloadUrl;
            // 期望下载的账单文件路径(当前目录下 mchCode+accountType+bill_date.csv)
            downloadBillRequest.localFilePath = mchCode + "_" + getFundFlowBillRequest.accountType.name() + "_" + getFundFlowBillRequest.billDate + ".csv";
            downloadBillRequest.tarType = DownloadBill.TarType.valueOf(getFundFlowBillRequest.tarType.name());
            downloadBillRequest.expectedHashType = DownloadBill.HashType.valueOf(response.hashType.name());
            downloadBillRequest.expectedHashValue = response.hashValue;

            downloadBill.run(downloadBillRequest);

            System.out.println("File downloaded successfully! Local file path: " + downloadBillRequest.localFilePath);

        } catch (WXPayUtility.ApiException e) {
            // TODO: 请求失败，根据状态码执行不同的逻辑
            String errorCode = e.getErrorCode();
            if (errorCode != null && errorCode.equals("NO_STATEMENT_EXIST")) {
                // 错误：账单不存在
                // 解决方式：请检查当前商户号在指定日期内是否有资金变动
                // 描述：说明指定日期无账单文件生成。若无资金变动，则无需重试
            } else if (errorCode != null && errorCode.equals("STATEMENT_CREATING")) {
                // 错误：账单正在生成中
                // 解决方式：请检查当前商户号在指定日期内是否有资金变动
                // 描述：说明指定日期账单正在生成中。若无资金变动，则无需重试；若有资金变动，则在T+1日上午10点后再重新下载
            } else if (errorCode != null && errorCode.equals("SYSTEM_ERROR")) {
                // 错误：系统错误
                // 解决方式：稍后重试
                // 描述：微信支付系统失败，系统失败直接立即重试大概率还会是系统失败，建议等1分钟后再重试
            } else if (errorCode != null && errorCode.equals("FREQUENCY_LIMITED")) {
                // 错误：限频报错
                // 解决方式：稍后重试
                // 描述：接口频率限制，直接立即重试大概率还会是系统失败，建议等1分钟后再重试
            } else if (errorCode != null && errorCode.equals("SIGN_ERROR")) {
                // 错误：签名错误
                // 解决方式：检查平台商户证书序列号，证书私钥文件，公钥ID，公钥文件，同时确认下签名过程是不是按照微信支付的签名方式
                // 描述：签名报错，需要确认签名材料和签名流程是否正确
            } else if (errorCode != null && errorCode.equals("PARAM_ERROR")) {
                // 错误：参数错误
                // 解决方式：按照报错返回的message，重新输入请求参数
                // 描述：参数的类型，长度，或者必填选项没有填写等
            } else if (errorCode != null && errorCode.equals("INVALID_REQUEST")) {
                // 错误：请求非法，请求参数正确，但是不符合下载账单业务规则
                // 解决方式：根据具体message查看具体哪里不符合业务规则，如果可以修改参数达到符合业务规则的修改请求符合业务规则之后再重试
                // 描述：不符合业务规则的场景如：
                // - 账单时间超过三个月，不允许申请下载账单
            } else {
                // 其他类型错误：稍等一会后重试
            }
        }
    }
}
```
