通过此API，商圈商户/服务商可自行查询用户的商圈会员积分服务授权状态。

- 接口频率：150QPS

- 前置条件：商圈商户/服务商已开通支付即积分产品功能



## 接口说明

支持商户：【普通服务商】

请求方式：【GET】 `/v3/businesscircle/user-authorizations/{openid}`

请求域名：【主域名】https://api.mch.weixin.qq.com 使用该域名将访问就近的接入点

　　　　　【备域名】https://api2.mch.weixin.qq.com 使用该域名将访问异地的接入点，指引[点击查看](https://pay.weixin.qq.com/doc/v3/partner/4012082567.md)

## 请求参数

### Header 参数
| 参数名 | 类型 | 必填 | 描述 |
| --- |  --- | --- | --- |
| Authorization | string | true | 请参考 [签名认证](https://pay.weixin.qq.com/doc/v3/partner/4012365870.md#%E8%AF%B7%E6%B1%82%E6%8E%A5%E5%8F%A3%E5%90%8E%E7%AB%AF%E7%AD%BE%E5%90%8D) 生成认证信息 |
| Accept | string | true | 请设置为 `application/json` |
### Path 参数
| 参数名 | 类型 | 必填 | 描述 |
| --- |  --- | --- | --- |
| openid | string(64) | true | **【顾客OpenID】** 顾客授权时使用的小程序上的OpenID |
### Query 参数
| 参数名 | 类型 | 必填 | 描述 |
| --- |  --- | --- | --- |
| sub_mchid | string(64) | false | **【商圈商户ID】** 当以服务商模式管理商圈积分能力时，则要带上商圈商户ID，否则留空 |
| appid | string(128) | true | **【小程序AppID】** 顾客授权积分时使用的小程序的AppID |
## 应答参数

200 OK

| 参数名 | 类型 | 必填 | 描述 |
| --- |  --- | --- | --- |
| openid | string(64) | true | **【顾客OpenID】** 顾客授权时使用的小程序上的OpenID |
| authorize_state | string | true | **【授权状态】** 顾客授权商圈积分结果，1."UNAUTHORIZED":未授权 ，2."AUTHORIZED":已授权 ，3."DEAUTHORIZED":已取消授权<br>可选取值：<br><ul><li>`UNAUTHORIZED`: 顾客无授权商圈进行支付即积分的记录</li></ul><br><ul><li>`AUTHORIZED`: 顾客已授权商圈进行支付即积分的能力</li></ul><br><ul><li>`DEAUTHORIZED`: 顾客已取消授权商圈进行支付即积分的能力</li></ul> |
| authorize_time | string(32) | false | **【授权时间】** 顾客成功授权商圈积分的时间 |
| deauthorize_time | string(32) | false | **【取消授权时间】** 顾客关闭授权商圈积分的时间 |

func QueryUserAuthorization(config *wxpay_utility.MchConfig, request *QueryUserAuthorizationRequest) (response *UserAuthorization, err error) {
	const (
		host   = "https://api.mch.weixin.qq.com"
		method = "GET"
		path   = "/v3/businesscircle/user-authorizations/{openid}"
	)

	reqUrl, err := url.Parse(fmt.Sprintf("%s%s", host, path))
	if err != nil {
		return nil, err
	}
	reqUrl.Path = strings.Replace(reqUrl.Path, "{openid}", url.PathEscape(*request.Openid), -1)
	query := reqUrl.Query()
	if request.SubMchid != nil {
		query.Add("sub_mchid", *request.SubMchid)
	}
	if request.Appid != nil {
		query.Add("appid", *request.Appid)
	}
	reqUrl.RawQuery = query.Encode()
	httpRequest, err := http.NewRequest(method, reqUrl.String(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Accept", "application/json")
	httpRequest.Header.Set("Wechatpay-Serial", config.WechatPayPublicKeyId())
	authorization, err := wxpay_utility.BuildAuthorization(config.MchId(), config.CertificateSerialNo(), config.PrivateKey(), method, reqUrl.RequestURI(), nil)
	if err != nil {
		return nil, err
	}
	httpRequest.Header.Set("Authorization", authorization)

	client := &http.Client{}
	httpResponse, err := client.Do(httpRequest)
	if err != nil {
		return nil, err
	}
	respBody, err := wxpay_utility.ExtractResponseBody(httpResponse)
	if err != nil {
		return nil, err
	}
	if httpResponse.StatusCode >= 200 && httpResponse.StatusCode < 300 {
		// 2XX 成功，验证应答签名
		err = wxpay_utility.ValidateResponse(
			config.WechatPayPublicKeyId(),
			config.WechatPayPublicKey(),
			&httpResponse.Header,
			respBody,
		)
		if err != nil {
			return nil, err
		}
		response := &UserAuthorization{}
		if err := json.Unmarshal(respBody, response); err != nil {
			return nil, err
		}

		return response, nil
	} else {
		return nil, wxpay_utility.NewApiException(
			httpResponse.StatusCode,
			httpResponse.Header,
			respBody,
		)
	}
}
